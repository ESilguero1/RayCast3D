17#include <stdint.h>
18
19// Fixed-point type: Q16.16 format
20typedef int32_t fixed_t;
21
22// Number of fractional bits
23#define FIXED_SHIFT 16
24#define FIXED_ONE (1 << FIXED_SHIFT) // 1.0 in fixed-point = 65536
25#define FIXED_HALF (1 << (FIXED_SHIFT-1)) // 0.5 in fixed-point = 32768
26
27// Conversion macros
28#define INT_TO_FIXED(x) ((fixed_t)((x) << FIXED_SHIFT))
29#define FIXED_TO_INT(x) ((int32_t)((x) >> FIXED_SHIFT))
30#define FLOAT_TO_FIXED(x) ((fixed_t)((x) * FIXED_ONE))
31#define FIXED_TO_FLOAT(x) ((double)(x) / FIXED_ONE)
32
33// Get fractional part (for texture coordinates)
34#define FIXED_FRAC(x) ((x) & (FIXED_ONE - 1))
35
36// Fixed-point multiplication: (a * b) >> 16
37// Uses 64-bit intermediate to avoid overflow
38static inline fixed_t fixed_mul(fixed_t a, fixed_t b) {
39 return (fixed_t)(((int64_t)a * b) >> FIXED_SHIFT);
40}
41
42// Fixed-point division: (a << 16) / b
43// Uses 64-bit intermediate to maintain precision
44static inline fixed_t fixed_div(fixed_t a, fixed_t b) {
45 return (fixed_t)(((int64_t)a << FIXED_SHIFT) / b);
46}
47
48// Fast absolute value
49static inline fixed_t fixed_abs(fixed_t x) {
50 return (x < 0) ? -x : x;
51}
52
53// Fixed-point floor (round toward negative infinity)
54static inline fixed_t fixed_floor(fixed_t x) {
55 return x & ~(FIXED_ONE - 1);
56}
57
58// Fixed-point ceiling
59static inline fixed_t fixed_ceil(fixed_t x) {
60 return (x + FIXED_ONE - 1) & ~(FIXED_ONE - 1);
61}
62

65#define FIXED_2PI 411775 // 2*PI * 65536

67#define FIXED_DEG_TO_RAD 1144 // (PI/180) * 65536 = 0.01745... * 65536
